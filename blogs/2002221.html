C语言实现简单的排序算法

冒泡排序

依次比较相邻两元素，若前一元素大于后一元素则交换之，直至最后一个元素即为最大；然后重新从首元素开始重复同样的操作，直至倒数第二个元素即为次大元素；依次类推。如同水中的气泡，依次将最大或最小元素气泡浮出水面。


                    #include <stdio.h>
                    #include <stdlib.h>
                    /*冒泡排序*/

                    void getNumber(int arr[], int n);
                    void bubbleSort(int arr[], int n);
                    void outPutNumber(int arr[], int n);
                    int main()
                    {
                        int n=1;
                        int arr[n];
                        printf("Please enter the number of array.\n");
                        scanf("%d", &n);
                        getNumber(arr, n);
                        bubbleSort(arr, n);
                        outPutNumber(arr, n);
                        return 0;
                    }
                    void getNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("Please enter the %d number.\n", i);
                            scanf("%d", &arr[i]);
                        }
                    }
                    void bubbleSort(int arr[], int n)
                    {
                        int i, k;
                        int mid;
                        for(i=n-1; i>-1; i--)
                        {
                            for(k=0; k<=i; k++)
                            {
                                if(arr[k]>arr[k+1])
                                {
                                    mid = arr[k];
                                    arr[k] = arr[k+1];
                                    arr[k+1] = mid;
                                }
                            }
                        }
                    }
                    void outPutNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("%5d", arr[i]);
                        }
                    }

选择排序

首先初始化最小元素索引值为首元素，依次遍历待排序数列，若遇到小于该最小索引位置处的元素则刷新最小索引为该较小元素的位置，直至遇到尾元素，结束一次遍历，并将最小索引处元素与首元素交换；然后，初始化最小索引值为第二个待排序数列元素位置，同样的操作，可得到数列第二个元素即为次小元素；以此类推。


                    #include <stdio.h>
                    #include <stdlib.h>
                    /*选择排序*/

                    void getNumber(int arr[], int n);
                    void selectSort(int arr[], int n);
                    void outPutNumber(int arr[], int n);
                    int main()
                    {
                        int n=1;
                        int arr[n];
                        printf("Please enter the number of array.\n");
                        scanf("%d", &n);
                        getNumber(arr, n);
                        selectSort(arr, n);
                        outPutNumber(arr, n);
                        return 0;
                    }
                    void getNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("Please enter the %d number.\n", i);
                            scanf("%d", &arr[i]);
                        }
                    }
                    void selectSort(int arr[], int n)
                    {
                        int i, k;
                        int mid, temp;
                        for(i=0; i<n; i++)
                        {
                            mid = i;
                            for(k=i+1; k<n; k++)
                            {
                                if(arr[mid]<arr[k])
                                {
                                    mid = k;
                                }
                            }
                            if(mid != i)
                            {
                                temp = arr[mid];
                                arr[mid] = arr[i];
                                arr[i] = temp;
                            }
                        }
                    }
                    void outPutNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("%5d", arr[i]);
                        }
                    }

插入排序

数列前面部分看为有序，依次将后面的无序数列元素插入到前面的有序数列中，初始状态有序数列仅有一个元素，即首元素。在将无序数列元素插入有序数列的过程中，采用了逆序遍历有序数列，相较于顺序遍历会稍显繁琐，但当数列本身已近排序状态效率会更高。


                    #include <stdio.h>
                    #include <stdlib.h>
                    /*插入排序*/

                    void getNumber(int arr[], int n);
                    void insertSort(int arr[], int n);
                    void outPutNumber(int arr[], int n);
                    int main()
                    {
                        int n=1;
                        int arr[n];
                        printf("Please enter the number of array.\n");
                        scanf("%d", &n);
                        getNumber(arr, n);
                        insertSort(arr, n);
                        outPutNumber(arr, n);
                        return 0;
                    }
                    void getNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("Please enter the %d number.\n", i);
                            scanf("%d", &arr[i]);
                        }
                    }
                    void insertSort(int arr[], int n)
                    {
                        int i, j, x;
                        for (i=1; i < n; i++)
                        {
                            if(arr[i]<arr[i-1])
                            {
                                j = i-1;
                                x = arr[i];
                                while(j>-1 && x<arr[j])
                                {
                                    arr[j+1]=arr[j];
                                    j--;
                                }
                                arr[j+1] = x;
                            }
                        }
                    }
                    void outPutNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("%5d", arr[i]);
                        }
                    }

归并排序

采用了分治和递归的思想，递归&分治-排序整个数列如同排序两个有序数列，依次执行这个过程直至排序末端的两个元素，再依次向上层输送排序好的两个子列进行排序直至整个数列有序（类比二叉树的思想，from down to up）


                    #include <stdio.h>
                    #include <stdlib.h>
                    /*归并排序*/

                    void getNumber(int arr[], int n);
                    void outPutNumber(int arr[], int n);
                    void mergeSort(int arr[], int p, int q, int r);
                    void merge(int arr[], int p, int r);
                    int main()
                    {
                        int n=1;
                        int arr[n];
                        printf("Please enter the number of array.\n");
                        scanf("%d", &n);
                        getNumber(arr, n);
                        merge(arr, 0, n-1);//合并排序
                        outPutNumber(arr, n);
                        return 0;
                    }
                    void getNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("Please enter the %d number.\n", i);
                            scanf("%d", &arr[i]);
                        }
                    }
                    void mergeSort(int arr[], int p, int q, int r)
                    {
                        int i, j, k, n1, n2;
                        int *front, *back;//变量用于申请两个内存空间
                        n1 = q-p+1;//前一部分长度
                        n2 = r-q;//后一部分长度

                        front = (int *)malloc(n1 * sizeof(int));//申请两个空间
                        back = (int *)malloc(n2 * sizeof(int));
                        /*将数组转入两个新空间中*/
                        for (i = 0; i < n1; i++)
                        {
                            front[i] = arr[p+i];
                        }
                        for (i = 0; i < n2; i++)
                        {
                            back[i] = arr[q+i+1];
                        }
                        /*将元素合并*/
                        i=0;
                        j=0;
                        k=p;
                        while (i<n1 && j<n2)
                        {
                            if (front[i]<back[j])
                            {
                                arr[k++] = front[i++];
                            }
                            else
                            {
                                arr[k++] = back[j++];
                            }
                        }
                        /*将剩余的元素合并*/
                        while (i<n1)
                        {
                            arr[k++] = front[i++];
                        }
                        while (j<n2)
                        {
                            arr[k++] = back[j++];
                        }

                    }
                    void merge(int arr[], int p, int r)
                    {
                        int q;
                        if (p<r)
                        {
                            q = (p+r)/2;            //将数组分解
                            merge(arr, p, q);       //将前面的一部分用递归方法排序
                            merge(arr, q+1, r);     //将后面的一部分用递归方法排序
                            mergeSort(arr, p, q, r);//将两个部分合并
                        }
                    }
                    void outPutNumber(int arr[], int n)
                    {
                        int i;
                        for(i=0; i<n; i++)
                        {
                            printf("%5d", arr[i]);
                        }
                    }

数组

利用指针来对数组元素进行移位操作

获得数组首地址后通过指针来对数组内部的元素进行操作

#include <stdio.h>

void move(int array[], int n, int m)
{
    int *p, array_end;

    array_end = *(array + n - 1);
    for (p = array + n - 1; p > array; p--)
    {
        *p = *(p - 1);
    }
    *array = array_end;
    m--;
    if (m > 0)
    {
        move(array, n, m);
    }
}

int main(void)
{
    int number[20], n, m, i;

    printf("共有多少个数？");
    scanf("%d", &n);
    printf("后移多少个？");
    scanf("%d", &m);
    printf("请以,号为间隔输入%d个数。\n", n);
    for (i = 0; i < n - 1; i++)
    {
        scanf("%d,", &number[i]);
    }
    scanf("%d", &number[n - 1]);
    move(number, n, m);
    printf("移动后顺序为：\n");
    for (i = 0; i < n - 1; i++)
    {
        printf("%d,", number[i]);
    }
    printf("%d", number[n - 1]);

    return 0;
}
